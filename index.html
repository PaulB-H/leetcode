Remove Duplicates from Sorted Array

Array, Easy

Solution:

var removeDuplicates = function (nums) {
  let lastNum;

  for (let i = 0; i < nums.length; i++) {
    if (nums[i] === lastNum) {
      nums.splice(i, 1);
      i--;
    } else {
      lastNum = nums[i];
    }
  }
  console.log(nums);
};
